// ── ACTUALIZAR CAMPO ── function actualizar(data) { const { usuario, campo, valor } = data; const sheet = iniciarSheet(); const rows = sheet.getDataRange().getValues(); const cols = { activo: 6, plan: 4, tienda_url: 5, vencimiento: 7 }; for (let i = 1; i < rows.length; i++) { if (rows[i][0] === usuario) { sheet.getRange(i + 1, cols[campo]).setValue(valor); // Si cambia plan, recalcular vencimiento if (campo === 'plan') { let dias = 0; if (valor === 'prueba') dias = 7; if (valor === 'junior') dias = 30; const vence = new Date(); if (dias > 0) { vence.setDate(vence.getDate() + dias); sheet.getRange(i + 1, 7).setValue(vence.toISOString().split("T")[0]); } else { sheet.getRange(i + 1, 7).setValue("9999-12-31"); } } return resp({ ok: true }); } } return resp({ ok: false, msg: 'Usuario no encontrado.' }); }
